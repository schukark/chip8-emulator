3use thiserror::Error;

5/// Chip8 ram struct
6#[derive(Debug, Clone, PartialEq, Eq)]
7pub struct Memory {
8    /// Bytes in memory
9    data: [u8; 4096],
10}
11
12use tklog::{error, trace};
13
14/// Predefined sprites for all hex digits
15const DIGIT_SPRITES: [[u8; 5]; 16] = [
16    [0xF0, 0x90, 0x90, 0x90, 0xF0], // ZERO
17    [0x20, 0x60, 0x20, 0x20, 0x70], // ONE
18    [0xF0, 0x10, 0xF0, 0x80, 0xF0], // TWO
19    [0xF0, 0x10, 0xF0, 0x10, 0xF0], // THREE
20    [0x90, 0x90, 0xF0, 0x10, 0x10], // FOUR
21    [0xF0, 0x80, 0xF0, 0x10, 0xF0], // FIVE
22    [0xF0, 0x80, 0xF0, 0x90, 0xF0], // SIX
23    [0xF0, 0x10, 0x20, 0x40, 0x40], // SEVEN
24    [0xF0, 0x90, 0xF0, 0x90, 0xF0], // EIGHT
25    [0xF0, 0x90, 0xF0, 0x10, 0xF0], // NINE
26    [0xF0, 0x90, 0xF0, 0x90, 0x90], // A
27    [0xE0, 0x90, 0xE0, 0x90, 0xE0], // B
28    [0xF0, 0x80, 0x80, 0x80, 0xF0], // C
29    [0xE0, 0x90, 0x90, 0x90, 0xE0], // D
30    [0xF0, 0x80, 0xF0, 0x80, 0xF0], // E
31    [0xF0, 0x80, 0xF0, 0x80, 0x80], // F
32];
33
34/// Enum with variants encoding all memory-related errors
35#[derive(Debug, Error)]
36#[must_use]
37pub enum MemoryError {
38    #[error("Reserved memory access")]
39    /// Addresses before 0x200 are reserved and can't be overwritten
40    PermissionDenied,
41    #[error("Sprite index out of range")]
42    /// There are exactly 16 sprites (0..=F), accesing other indices is erroneous
43    IncorrectSprite,
44    #[error("Memory access out of range: {0:#X}")]
45    /// Access of memory beyond 12 bit of addresses available
46    OutOfRange(u16),
47}
48
49impl Memory {
50    /// Create a ram filled with digit sprites (0x000 - 0x1FF)
51    pub fn new() -> Self {
52        let mut data = [0; 4096];
53
54        for (number, constant) in DIGIT_SPRITES.iter().enumerate() {
55            for (i, byte) in constant.iter().enumerate() {
56                data[number * 5 + i] = *byte;
57            }
58        }
59
60        Self { data }
61    }
62
63    /// Fetch sprite address from reserved memory
64    ///
65    /// Returns error if the sprite asked is not in 0..=F
66    pub fn read_sprite_address(&self, digit: u8) -> Result<u16, MemoryError> {
67        if digit > 0xF {
68            error!(
69                "Incorrect digit sprite asked, should be <= 0xF, was ",
70                digit
71            );
72            Err(MemoryError::IncorrectSprite)
73        } else {
74            trace!("Found digit ", digit, " sprite at ", digit as u16 * 5);
75            Ok(digit as u16 * 5)
76        }
77    }
78
79    /// Fetch a byte from address
80    pub fn read_byte(&self, addr: u16) -> Result<u8, MemoryError> {
81        if addr >= 1 << 12 {
82            error!("Got incorrect address, should be below 4096, was ", addr);
83            return Err(MemoryError::OutOfRange(addr));
84        }
85        trace!(
86            "Read byte ",
87            self.data[addr as usize], " from address ", addr
88        );
89        Ok(self.data[addr as usize])
90    }
91
92    /// Fetch a 2-byte word from address
93    pub fn read_word(&self, addr: u16) -> Result<u16, MemoryError> {
94        if addr >= (1 << 12) - 1 {
95            error!("Got incorrect address, should be below 4095, was ", addr);
96            return Err(MemoryError::OutOfRange(addr));
97        }
98        let hi = self.data[addr as usize] as u16;
99        let lo = self.data[(addr + 1) as usize] as u16;
100        trace!("Read word ", (hi << 8) | lo, " from address ", addr);
101        Ok((hi << 8) | lo)
102    }
103
104    /// Load bytes into ram starting from given address
105    pub fn load(&mut self, start: u16, bytes: &[u8]) -> Result<(), MemoryError> {
106        if start < 0x200 {
107            error!(
108                "Requested reserved memory, addresses should be at leas 512, got ",
109                start
110            );
111            return Err(MemoryError::PermissionDenied);
112        }
113
114        if start as usize + bytes.len() > self.data.len() {
115            error!(
116                "Invalid memory request, address should be below 4096, got ",
117                start + bytes.len() as u16
118            );
119            return Err(MemoryError::OutOfRange(start + bytes.len() as u16));
120        }
121
122        self.data[start as usize..start as usize + bytes.len()].copy_from_slice(bytes);
123        trace!(format!(
124            "Wrote {bytes:?} to memory from {start} to {}",
125            start + bytes.len() as u16
126        ));
127        Ok(())
128    }
129}

144    #[test]
145    fn test_incorrect_sprite_index() {
146        let memory = Memory::new();
147
148        assert!(matches!(
149            memory.read_sprite_address(0xF + 2),
150            Err(MemoryError::IncorrectSprite)
151        ));
152    }
153
154    #[test_case(0x0 ; "lowest sprite")]
155    #[test_case(0xF ; "highest sprite")]
156    #[test_case(0x8 ; "random sprite")]
157    fn test_correct_sprite_indexes(index: u8) {
158        let memory = Memory::new();
159
160        // 5 rows for each digit
161        assert_eq!(memory.read_sprite_address(index).unwrap(), index as u16 * 5);
162    }
163
164    #[test]
165    fn test_read_byte_out_of_range() {
166        let memory = Memory::new();
167
168        assert!(matches!(
169            memory.read_byte(1 << 12),
170            Err(MemoryError::OutOfRange(_))
171        ));
172    }
173
174    #[test_case(0x200 ; "lowest address")]
175    #[test_case(0xFFF ; "highest address")]
176    #[test_case(0xB42 ; "random address")]
177    fn test_read_byte_correct(addr: u16) {
178        let memory = Memory::new();
179
180        assert_eq!(memory.read_byte(addr).unwrap(), 0);
181    }

184    fn test_read_word_out_of_range() {
185        let memory = Memory::new();

188            memory.read_word(1 << 12),
189            Err(MemoryError::OutOfRange(_))
190        ));

194            memory.read_word((1 << 12) - 1),

197    }
