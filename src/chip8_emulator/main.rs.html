17#[cfg_attr(coverage_nightly, coverage(off))]
18#[allow(clippy::borrow_interior_mutable_const)] // As per docs of tklog, this is correct
19/// Setup logging
20fn log_init() {
21    LOG.set_console(false) // Enables console logging
22        .set_level(LEVEL::Info) // Sets the log level; default is Debug
23        .set_format(Format::LevelFlag | Format::Time | Format::ShortFileName) // Defines structured log output with chosen details
24        .set_cutmode_by_size("logs/tklogsize.txt", 1 << 22, 10, true) // Cuts logs by file size (4 MB), keeps 10 backups, compresses backups
25        .set_formatter("{level}{time} {file}:{message}\n"); // Customizes log output format; default is "{level}{time} {file}:{message}"
26}
27
28#[cfg_attr(coverage_nightly, coverage(off))]
29fn main() {
30    let args = std::env::args();
31    log_init();
32
33    if args.len() != 2 {
34        eprintln!("Usage: chip8-emulator /path/to/your/rom");
35        return;
36    }
37
38    run_app(args).expect("Error occured when running the application");
39}
